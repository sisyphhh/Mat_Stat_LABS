def pus(masss):
    y=[]
    e=2.718
    liambda=1.47
    
    for i in range(len(masss)):
       
        answww=1
        
        indx=masss[i]
        
        if indx==0:
            answww=(1/e)**liambda
            answww = float('{:.5f}'.format(answww))
            y.append(answww)         
        else :    
            for j1 in range(indx):
                answww*=liambda/(j1+1) 
            answww*=(1/e)**liambda
            answww = float('{:.5f}'.format(answww))
            y.append(answww) 
    
    return y
